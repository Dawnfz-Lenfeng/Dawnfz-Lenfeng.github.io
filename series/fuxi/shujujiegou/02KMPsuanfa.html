	n = text[0]
	m = pattern[0]
	j = 1

	for i in range(1, n + 1):
		while j > 0 and text[i] != pattern[j]:
			j = next[j]

		# 如果 text[i]==pattern[j], 此时 i, j 依次递增即可

def kmp_search(text: str, pattern: str, next: list[int]):
	n = len(text)
	m = len(pattern)
	j = 0

	for i in range(n):
		while j >= 0 and text[i] != pattern[j]:
			j = next[j]

		j += 1
		if j == m:
			return i - m + 1

	return -1

def get_next(pattern) -> list[int]:
	m = len(pattern)
	next = [0] * m
	next[0] = k = -1

	for i in range(1, m):
		while k >= 0 and pattern[i - 1] != pattern[k]:
			k = next[k]

		k += 1
		next[i] = k

	return next
